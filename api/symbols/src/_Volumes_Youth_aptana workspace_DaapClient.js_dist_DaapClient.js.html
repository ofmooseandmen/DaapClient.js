 19 /**
 20  * DAAP Client. This client connects to the DAAP server at the specified IP address/port.
 21  * <p>
 22  * Since DaapClient.js relies on XmlHttpRequest to connect to the DAAP server, this latter shall
 23  * allow Cross-Origin Ressource Sharing. If you cannot modify your DAAP server to include Access-Control-Allow-Origin
 24  * in the response header, consider setting up a proxy such as (apache configuration file example):
 25  *
 26  * <p>
 27  * <pre>
 28  *"Listen 3690
 29  *
 30  * NameVirtualHost *:3690
 31  *
 32  * <VirtualHost *:3690>
 33  *   ProxyRequests On
 34  *   <Proxy>
 35  *     Order deny,allow
 36  *     Allow from all
 37  *   </Proxy>
 38  *
 39  *   ProxyPass / http://localhost:3689/
 40  *   ProxyPassReverse / http://localhost:3689/
 41  *   Header set Access-Control-Allow-Origin "*"
 42  *   Header set Access-Control-Allow-Methods "GET
 43  * </VirtualHost>"
 44  * </pre>
 45  * In which case DaapClient shall be constructed with port number 3690
 46  *
 47  * <p>
 48  * Example of use:
 49  * <pre>
 50  * // create the DaapClient, DAAP server IP address = 10.0.1.6; port = 3690.
 51  * var client = new DaapClient("10.0.1.6", 3690);
 52  *
 53  * var streamsFetched = function streamsFetched(code, streams) {
 54  *     if (code == 200) {
 55  *         alert(streams[0].uri);
 56  *     } else {
 57  *         alert("Could not fetch streams: [HTML Status code = " + code + "]");
 58  *     }
 59  * };
 60  *
 61  * var loginCompleted = function loginCompleted(code) {
 62  *     if (code == 200) {
 63  *         client.fetchStreams(streamsFetched);
 64  *     } else if (code == 401) {
 65  *         var passwd = prompt("Enter Password for DAAP server");
 66  *         client.secureLogin(passwd, loginCompleted);
 67  *     } else {
 68  * 	       alert("Could not login to the DAAP server: [HTML Status code = " + code + "]");
 69  *     }
 70  * };
 71  *
 72  * // start with unsecure login - no password.
 73  * client.login(loginCompleted);
 74  * </pre>
 75  *
 76  * @see http://www.w3.org/wiki/CORS_Enabled
 77  *
 78  * @constructor
 79  * @param ip {String} the IP address of the DAAP Server
 80  * @param port {int} the port of the DAAP Server - if omitted, 3689 is assumed
 81  */
 82 
 83 function DaapClient(ip, port) {
 84 
 85     /** @private the DAAP server. */
 86     var server;
 87     // if port not provided; port = 3689.
 88     if (typeof(port) == 'undefined') {
 89         server = "http://" + ip + ":/3689";
 90     } else {
 91         server = "http://" + ip + ":" + port;
 92     }
 93 
 94     /** @private the HTTP client to communicate with the server. */
 95     var httpClient = new DaapHttpClient(server);
 96 
 97     /** @private the session id. */
 98     var sid = null;
 99 
100     /** @private the revision id. */
101     var rid = null;

105      *
106      * @constructor
107      */
108 
109     function EndOfPacketException() {
110

113     /**
114      * The DAAP response definition: a packet organized by 'codes' and 'values'. The basic format of DAAP is:
115      *
116      * <pre>
117      *   4 byte tag | size (4 byte int) | data
118      * </pre>
119      *
120      * where the length of the data is determined by the size. The 4 byte tag can be represented as either an integer,
121      * or as a four character string.
122      * <p>
123      * The data can itself be either:
124      * <ul>
125      * <li>the actual transmitted data: an <code>int</code> or a <code>String</code>
126      * <li>a sequence of chunks (from <code>1<code> to <code>n</code>)
127      * </ul>
128      *
129      * @see <a href="http://tapjam.net/daap">DAAP Protocol documentation v0.2</a>
130      *
131      * @constructor
132      * @param chunk {String}
133      *            a string representing a DAAP chunk (code + size +
134      *            data).
135      * @param start {int}
136      *            the offset at which to start reading the chunk - if omitted, 0 is assumed
137      * @throws EndOfPacketException
138      *             if the string does not hold enough information to
139      *             extract code + data.
140      */
141 
142     function DaapPacket(chunk, start) {
143 
144         /** @private size in bytes of "code" of chunk. */
145         var CODE_LENGTH = 4;
146 
147         /** @private size in bytes of "size" of chunk. */
148         var SIZE_LENGTH = 4;
149 
150         /** @private size in bytes of "header" of chunk. */
151         var HEADER_LENGTH = CODE_LENGTH + SIZE_LENGTH;
152 
153         // if start is not provided, start = 0.
154         if (typeof(start) == 'undefined') {
155             start = 0;
156         }
157 
158         /** @private the current offset - initialize @ 0; used to read through the chunk. */
159         var offset = 0;
160         var length = chunk.length - start;
161         // check data holds at least 8 bytes; header + size:
162         if (length < HEADER_LENGTH) {
163             throw new EndOfPacketException();
164         }
165 
166         // first 4 bytes is the DAAP code of this chunk.
167         var code = chunk.substring(start, start + CODE_LENGTH);
168 
169         // next 4 bytes is the size
170         var size = readUInt32(chunk, start + CODE_LENGTH);
171 
172         // check data holds at least computed size
173         if (length < HEADER_LENGTH + size) {
174             throw new EndOfPacketException();
175 
176         }
177 
178         /** @private the binary data that hold the content of this packet. */
179         // data is next spanning over 'size'. the byte array holding the data associated to the DAAP code.
180         var data = chunk.substring(start + HEADER_LENGTH, start + HEADER_LENGTH + size);
181 
182         /**
183          * Read the next chunk within this chunk.
184          * <p>
185          * Note: it is up to the user to use this method as long as the data associated to the returned chunk do contain
186          * yet other chunks. Once the return chunk contains the actual transmitted data use
187          * {@link DaapPacket#convertToInt()} or {@link DaapPacket#convertToString()}. Calling this method on an actual
188          * data will throw a {@link EndOfPacketException}.
189          *
190          * @return the next read chunk
191          * @throws EndOfPacketException if no next chunk can be read.
192          */
193         this.readNextChunk = function() {
194             var result = new DaapPacket(data, offset);
195             offset += result.size();
196             return result;
197         };
198 
199         /**
200          * Return <code>true</code> if the specified code equals this code. Comparison is made at byte level.
201          *
202          * @param other {String} the specified code
203          * @return <code>true</code> if the specified String equals this code
204          */
205         this.codeEquals = function(other) {
206             return code == other;
207         };
208 
209         /**
210          * Return the size of this chunk including 'code' and 'size'.
211          *
212          * @return the size of this chunk
213          */
214         this.size = function() {
215             return data.length + HEADER_LENGTH;
216         };
217 
218         /**
219          * Convert the data associated to this packet into an <code>int</code>.
220          *
221          * @return the data associated to this packet converted into an <code>int</code>.
222          */
223         this.convertToInt = function() {
224             return readUInt32(data, 0);
225         };
226 
227         /**
228          * Convert the data associated to this packet into a <code>String</code>.
229          *
230          * @return the data associated to this packet converted into a <code>String</code>.
231          */
232         this.convertToString = function() {
233             return data;
234         };
235 
236         /**
237          * Seek this packet for the specified code and return first packet with matching code.
238          *
239          * @param refCode {String} the specified code
240          * @return the first {DaapPacket} which code matches the specified one.
241          */
242         this.seekFirst = function(refCode) {
243             // reset offset to seek for specified code throughout all data.
244             offset = 0;
245             var result = null;
246             while (true) {
247                 try {
248                     result = this.readNextChunk();
249                     if (result.codeEquals(refCode)) {
250                         break;
251                     }
252                 } catch (e) {
253                     if (e instanceof EndOfPacketException) {
254                         result = null;
255                         break;
256                     } else {
257                         throw e;
258                     }
259                 }
260             }
261             return result;
262         };
263 
264         /**
265          * Seek this packet for the specified code and return all packets with matching code.
266          *
267          * @param refCode {String} the specified code
268          * @return an array of {DaapPacket} which code matches the specified one.
269          */
270         this.seek = function(refCode) {
271             // reset offset to seek for specified code throughout all data.
272             offset = 0;
273             var result = [];
274             var next;
275             while (true) {
276                 try {
277                     next = this.readNextChunk();
278                     if (next.codeEquals(refCode)) {
279                         result.push(next);
280                     }
281                 } catch (e) {
282                     if (e instanceof EndOfPacketException) {
283                         break;
284                     } else {
285                         throw e;
286                     }
287                 }
288             }
289             return result;
290         };
291 
292         /**
293          * Read big-endian (network byte order) unsigned 32-bit <code>int</code> from data at offset
294          *
295          * @see http://fhtr.blogspot.com/2009/12/3d-models-and-parsing-binary-data-with.html
296          *
297          * @param data {String} data to read from
298          * @param offset {int} offset
299          * @return the read unsigned 32-bit
300          */
301 
302         function readUInt32(data, offset) {
303             return ((data.charCodeAt(offset) & 0xFF) << 24) + ((data.charCodeAt(offset + 1) & 0xFF) << 16) + ((data.charCodeAt(offset + 2) & 0xFF) << 8) + (data.charCodeAt(offset + 3) & 0xFF);
304         }
305 
306     }

308     /**
309      * The DAAP HTTP client based on XMLHttpRequest.
310      *
311      * @constructor
312      * @param aServer {String} the DAAP Server.
313      */
314 
315     function DaapHttpClient(aServer) {
316 
317         /** @private the DAAP server. */
318         var server = aServer;
319 
320         /** @private the base 64 encoded password for authentication. */
321         var encodedPassword = null;
322 
323         /**
324          * Execute the specified request.
325          * <p>
326          * request shall provide handleResponse and getUri methods.
327          *
328          * @see http://badassjs.com/post/694057326/binaryreader-js-parsing-binary-data-in-javascript
329          * @param request {Object} the request to execute.
330          */
331         this.execute = function(request) {
332             var xhr = new XMLHttpRequest();
333             xhr.onreadystatechange = function() {
334                 if (this.readyState == 4) {
335                     if (this.status != 200) {
336                         request.fail(this.status);
337                     } else {
338                         try {
339                             var packet = new DaapPacket(xhr.responseText);
340                             request.handleResponse(packet);
341                         } catch (e) {
342                             if (e instanceof EndOfPacketException) {
343                                 request.fail(400);
344                             } else {
345                                 throw e;
346                             }
347                         }
348                     }
349                 }
350             };
351 
352 
353             xhr.open("GET", "http://" + ip + ":" + port + "/" + request.getUri(), true);
354 
355             // The following line says we want to receive data as Binary and not as Unicode
356             if (!('overrideMimeType' in xhr)) {
357                 throw new Error("Your browser does not support binary encoding, aborting ...");
358             }
359             xhr.overrideMimeType('text/plain; charset=x-user-defined');
360             // add basic authentication is password provided.
361             if (encodedPassword != null) {
362                 xhr.setRequestHeader('Authorization', 'Basic ' + encodedPassword);
363             }
364             xhr.send();
365         };
366 
367         /**
368          * Set the password use for HTTP authentication.
369          *
370          * @param password the password - plain text (not base64 encoded)
371          */
372         this.setPassword = function(password) {
373             // use built-in base64 encoder, maybe one day IE will implement it...
374             encodedPassword = btoa("admin:" + password);
375         };
376 
377     }

379     /**
380      * Login request handler; retrieves session ID and is a callback for {DaapHttpClient#execute(request)}.
381      *
382      * @constructor
383      * @param l {Object} the listener
384      */
385 
386     function LoginRequestHandler(l) {
387 
388         /** @private the listener. */
389         var listener = l;
390 
391         /**
392          * Handle the response of the DAAP server to the login request.
393          * <p>
394          * Fires sidUpdated event.
395          *
396          * @param packet {Object} the DAAP packet received from the server upon login request
397          */
398         this.handleResponse = function(packet) {
399             var sid = packet.seekFirst('mlid').convertToInt();
400             l.sidUpdated(sid);
401         };
402 
403 
404         this.fail = function(code) {
405             l.fail(code);
406         };
407 
408         /**
409          * Returns the login request URI.
410          *
411          * @return the login request URI
412          */
413         this.getUri = function() {
414             return "login";
415         };
416 
417     }

419     /**
420      * Update request handler; retrieves revision ID and is a callback for {DaapHttpClient#execute(request)}.
421      *
422      * @constructor
423      * @param l {Object} the listener
424      * @param aSid {String} the DAAP session ID
425      */
426 
427     function UpdateRequestHandler(aSid, l) {
428 
429         /** @private the DAAP session ID. */
430         var sid = aSid;
431 
432         /** @private the listener. */
433         var listener = l;
434 
435         /**
436          * Handle the response of the DAAP server to the update request.
437          * <p>
438          * Fires ridUpdated event.
439          *
440          * @param packet {Object} the DAAP packet received from the server upon update request

442         this.handleResponse = function(packet) {

444             l.ridUpdated(rid);

447

450         };
451

454          *

456          */

463     /**
464      * Database request handler; retrieves all the songs in the database of the DAAP server
465      *
466      * @constructor
467      * @param aSid {String} the DAAP session ID
468      * @param aRid {String} the DAAP revision ID
469      * @param aServer {String} the DAAP server IP address
470      * @param l {Object} the listener
471      */
472 
473     function DatabaseRequestHandler(aSid, aRid, aServer, aCallback) {
474 
475         /** @private the DAAP session ID. */
476         var sid = aSid;
477 
478         /** @private the DAAP revision ID. */
479         var rid = aRid;
480 
481         /** @private the DAAP server IP address. */
482         var server = aServer;
483 
484         /** @private the callback. */
485         var callback = aCallback;
486 
487         var fields = ["dmap.itemid", "daap.songformat", "dmap.itemname", "daap.songalbum", "daap.songartist", "daap.songgenre", "daap.songtracknumber"];
488 
489         /**
490          * Handle the response of the DAAP server to the database request.
491          * <p>
492          * Fires streamsFetched event.
493          *
494          * @param packet {Object} the DAAP packet received from the server upon database request
495          */
496         this.handleResponse = function(packet) {
497             var mlcl = packet.seekFirst("mlcl");
498             var mlits = mlcl.seek("mlit");
499             var mlitsLength = mlits.length;
500             var audioStreams = [];
501             for (var i = 0; i < mlitsLength; i++) {
502                 audioStreams.push(createDaapStream(mlits[i]));
503             }
504             callback(200, audioStreams);
505         };
506 
507 
508         this.fail = function(code) {
509             callback(code, undefined);
510         };
511 
512         /**
513          * Returns the database request URI.
514          *
515          * @return the database request URI
516          */
517         this.getUri = function() {
518             return "databases/1/items?type=music&session-id=" + sid + "&revision-id=" + rid + "&meta=" + fields.join();
519         };
520 
521         function createDaapStream(mlit) {
522             var daapSongId = extractInt(mlit, "miid");
523             var songFormat = extractString(mlit, "asfm");
524             var uri = server + "/databases/1/items/" + daapSongId + "." + songFormat + "?session-id=" + sid;
525             // song id is <session-id>-<song-id>
526             var id = sid + "-" + daapSongId;
527             var trackNumber = extractInt(mlit, "astn");
528             var title = extractString(mlit, "minm");
529             var album = extractString(mlit, "asal");
530             var artist = extractString(mlit, "asar");
531             var genre = extractString(mlit, "asgn");
532             var result = {
533                 uri: uri,
534                 trackNumber: trackNumber,
535                 title: title,
536                 album: album,
537                 artist: artist,
538                 genre: genre,
539                 id: id
540             };
541             return result;
542         }
543 
544         /**
545          * Extract the <code>int</code> value corresponding to the specified code.
546          *
547          * @param packet {DaapPacket} the DAAP chunk from which the extract the value.
548          * @param code {String} the specified code
549          * @return the <code>int</code> value corresponding to the specified code
550          */
551 
552         function extractInt(packet, code) {
553             var chunk = packet.seekFirst(code);
554             var result = -1;
555             if (chunk != null) {
556                 result = chunk.convertToInt();
557             }
558             return result;
559         }
560 
561         /**
562          * Extract the <code>String</code> value corresponding to the specified code.
563          *
564          * @param packet {DaapPacket} the DAAP chunk from which the extract the value.
565          * @param code {String} the specified code
566          * @return the <code>String</code> value corresponding to the specified code
567          */
568 
569         function extractString(packet, code) {
570             var chunk = packet.seekFirst(code);
571             var result = "";
572             if (chunk != null) {
573                 result = chunk.convertToString();
574             }
575             return result;
576         }
577 
578     }

580     /**
581      * The Login listener.
582      * <p>
583      * Login to a DAAP server is a two-step processing: first retrieve the Session ID (SID) and then the Revision ID (RID).
584      *
585      * @constructor
586      * @param aCallback the callback function to be called once login phase is completed (SID and RID retrieved)
587      */
588 
589     function LoginListener(aCallback) {
590 
591         /** @private the callback function. */
592         var callback = aCallback;
593 
594         /**
595          * Session ID updated event handler.
596          * <p>
597          * Once SID has been computed, RID shall be retrieved.
598          *
599          * @param aSid the updated SID
600          */
601         this.sidUpdated = function(aSid) {
602             sid = aSid;
603             // retrieve revision id.
604             var handler = new UpdateRequestHandler(sid, this);
605             httpClient.execute(handler);
606         };
607 
608         /**
609          * Revision ID updated event handler.
610          * <p>
611          * Both SID and RID have been computed, login phase is over.
612          *
613          * @param aRid the updated RID
614          */
615         this.ridUpdated = function(aRid) {
616             rid = aRid;
617             callback(200);
618         };
619 
620 
621         this.fail = function(code) {
622             callback(code);
623         };
624 
625     }
626 
627     /**
628      * If SID or RID is <code>null</code> throws Error.
629      *
630      * @private
631      */
632 
633     function checkLogin() {
634         if (sid == null || rid == null) {
635             throw new Error("Login not completed.");
636         }
637     }
638 
639     /**
640      * Log on to the DAAP server with the specified password.
641      *
642      * @param password the password to connect to the DAAP server - optional, if none provided no HTTP Authorization header will be sent
643      * @param callback the callback function called once the login phase is over. The HTML status code is returned.
644      */
645     this.secureLogin = function(password, callback) {
646         if (typeof(password) != 'undefined') {
647             httpClient.setPassword(password);
648         }
649         this.login(callback);
650     };
651 
652     /**
653      * Log on to the DAAP server.
654      *
655      * @param callback the callback function called once the login phase is over. The HTML status code is returned.
656      */
657     this.login = function(callback) {
658         var l = new LoginListener(callback);
659         // retrieve session id.
660         var handler = new LoginRequestHandler(l);
661         httpClient.execute(handler);
662     };
663 
664     /**
665      * Fetch all audio streams served by the DAAP server.
666      * <p>
667      * Each stream is described by the following JSON format:
668      * <ul>
669      * <li>uri : the address of the stream
670      * <li>trackNumber : the track number in the album
671      * <li>title : the stream title
672      * <li>album : the stream album
673      * <li>artist : the stream artist
674      * <li>genre : the stream genre
675      * <li>id : the DAAP ID of the stream
676      * <li>
677      * </ul>
678      * <p>
679      * @param callback the callback function called once the streams have been fetched. Callback is called with HTML status code and an array of stream or 'undefined' if the the status code is not <code>200</code>.
680      */
681     this.fetchStreams = function(callback) {
682         checkLogin();
683         var handler = new DatabaseRequestHandler(sid, rid, server, callback);
684         httpClient.execute(handler);
685     };
